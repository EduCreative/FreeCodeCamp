function pairElement(str) {
  const myString = [];

  for(let i=0; i<str.length; i++){
    if(str[i] === "G"){
      myString.push(["G", "C"])
    } else if(str[i] === "C"){
        myString.push(["C", "G"])
    } else if(str[i] === "A"){
        myString.push(["A", "T"])
    } else if(str[i] === "T"){
      myString.push(["T", "A"])
  }
  }
  console.log(myString);
  return myString;
}
